% Margin glosses for acronyms and jargon (\firstterm{key} on first use).
% Every gloss is emitted with the abbreviation in bold up front; keys whose
% display form differs from the key name get a \declaretermlabel override.
\makeatletter
\newcommand{\declaretermsnip}[2]{\expandafter\gdef\csname termsnip@#1\endcsname{#2}}
\newcommand{\declaretermlabel}[2]{\expandafter\gdef\csname termlabel@#1\endcsname{#2}}
\newcommand{\termsnip@note}[1]{%
  \marginnote{\textbf{\ifcsname termlabel@#1\endcsname
      \csname termlabel@#1\endcsname\else#1\fi}%
    \ \csname termsnip@#1\endcsname}%
}
\newcommand{\firstterm}[1]{%
  \expandafter\ifx\csname termused@#1\endcsname\relax
    \expandafter\gdef\csname termused@#1\endcsname{}%
    \ifcsname termsnip@#1\endcsname
      \termsnip@note{#1}%
    \fi
  \fi
}
% Unconditional variant: always emits the gloss, even if the term appeared
% earlier in the book. Used in standalone material like the interview appendix.
\newcommand{\termnote}[1]{%
  \ifcsname termsnip@#1\endcsname
    \termsnip@note{#1}%
  \fi
}
\makeatother

\declaretermlabel{FR8DR8}{FR8/DR8}
\declaretermlabel{AECQ100}{AEC-Q100}
\declaretermlabel{dspserdes}{DSP SerDes}
\declaretermlabel{gearbox}{Gearbox}

\declaretermsnip{FEC}{Forward error correction: parity symbols let the receiver fix
  bit errors without retransmit. Datacom links specify a \emph{pre-FEC} BER the code
  must correct.}

\declaretermsnip{KP4}{IEEE 802.3 Clause~91 FEC for PAM4: Reed--Solomon RS(544,514) on
  10-bit symbols, 5.5\% overhead. Pre-FEC BER target $2.4\times10^{-4}$.}

\declaretermsnip{RS544514}{Reed--Solomon block code: 544 symbols out, 514 payload, 30
  parity. Corrects up to 15 symbol errors per codeword.}

\declaretermsnip{preFEC}{BER measured before the FEC decoder. Standards and
  validation sweeps use this number, not post-FEC.}

\declaretermsnip{postFEC}{BER after FEC correction. Target is effectively
  error-free ($<10^{-12}$ class).}

\declaretermsnip{MLC}{Multi-level coding: coded modulation paired with PAM6/PAM8. OIF
  448G workshop models often add a strong outer RS code; $\sim$12\% overhead vs.\ KP4
  at 5.5\%.}

\declaretermsnip{HDFEC}{Hard-decision FEC: high-overhead codes used in optical demos and
  long-reach products (10--15\% OH), not the host KP4 budget.}

\declaretermsnip{OH}{Coding overhead: fraction of line rate spent on parity symbols
  (KP4 $\approx$5.5\%).}

\declaretermsnip{TDECQ}{Transmitter and dispersion eye closure quaternary:
  headline PAM4 transmitter-quality metric after a reference receiver and bounded
  FFE, including a test fiber. Lower is better; LPO MSA 100G-DR caps near 3.4~dB.
  Replaced simple eye-mask tests. Stressed counterpart: SECQ.}

\declaretermsnip{TECQ}{Transmitter eye closure quaternary: same measurement as
  TDECQ without the test fiber. Used with TDECQ in LPO MSA OMA tables.}

\declaretermsnip{SECQ}{Stressed eye closure quaternary: receiver-side stressed test,
  analog of TDECQ. Applies calibrated optical stress before BER test.}

\declaretermsnip{MDI}{Medium dependent interface: the optical connector where a module
  meets the fiber. TP2 (Tx launch) and TP3 (Rx input) sit here.}

\declaretermsnip{EECQ}{Electrical eye closure quaternary: the electrical analog of
  TDECQ, quoted at the host-referenced points TP1a and TP4a.}

\declaretermsnip{EAM}{Electro-absorption modulator: voltage-controlled absorption on InP;
  low chirp vs DML; paired with DFB in EML.}

\declaretermsnip{APD}{Avalanche photodiode: internal gain improves sensitivity; excess
  noise factor and bias complexity vs PIN.}

\declaretermsnip{TIA}{Transimpedance amplifier: converts PD current to voltage; input
  capacitance sets input-referred noise.}

\declaretermsnip{PIN}{PIN photodiode: no internal gain; Ge-on-Si mainstream at 100--224G.}

\declaretermsnip{gearbox}{Rate/format converter between electrical and optical lanes;
  common in 448G transition; incompatible with strict LPO.}

\declaretermsnip{CMIS}{Common Management Interface Specification: module management,
  monitors, link training at 224G/448G.}

\declaretermsnip{RMA}{Return merchandise authorization: field-failed unit returned for
  FA. Distinct failure codes keep fleet FIT honest.}

\declaretermsnip{ATP}{Acceptance test plan: measurable ship tests that prove the
  requirements contract with a supplier (often mapped to GR-468 stresses).}

\declaretermsnip{SPC}{Statistical process control: control charts on key parameters
  ($I_\mathrm{th}$, SMSR, power, burn-in fallout) to catch drift qual lots missed.}

\declaretermsnip{eightD}{Eight disciplines (8D) corrective action: contain, root-cause,
  correct, and prevent with the supplier; close only with verification evidence.}

\declaretermsnip{DPA}{Destructive physical analysis: cross-section / EDX on failed
  units to confirm facet, solder, or FAU failure modes.}

\declaretermsnip{LOS}{Loss of signal: receiver (or host) flag that optical input is
  below the detect threshold.}

\declaretermsnip{LOL}{Loss of lock: CDR or SerDes unlocked; electrical path not
  recovered even if light is present.}

\declaretermsnip{FAIR}{First-article inspection report: re-qualify after tooling, site,
  epi, or CMIS firmware change before open volume.}

\declaretermsnip{DML}{Directly modulated laser: modulate bias current; high chirp, low cost.}

\declaretermsnip{DFB}{Distributed-feedback laser: grating in the active region for
  single-mode output. Dominant 1310~nm source for DR/FR and EML gain sections.}

\declaretermsnip{VCSEL}{Vertical-cavity surface-emitting laser: 850~nm class, MMF SR links.}

\declaretermsnip{LIV}{Light--current--voltage curve: threshold, slope efficiency, kink-free
  range, and thermal rollover versus bias.}

\declaretermsnip{SMSR}{Side-mode suppression ratio: power difference (dB) between the lasing
  mode and the strongest side mode (OSA).}

\declaretermsnip{RIN}{Relative intensity noise: laser amplitude noise spectral density
  (dB/Hz). Specs often quote stressed $\mathrm{RIN}_x\mathrm{OMA}$ under fixed ORL.}

\declaretermsnip{APC}{Automatic power control: feedback loop from a monitor PD that
  holds average optical power; slow loop BW keeps it out of the RIN band.}

\declaretermsnip{ELSFP}{External Laser Small Form-Factor Pluggable (OIF): faceplate CW laser
  module for CPO; 24-pin card edge, CMIS, blind-mate MT.}

\declaretermsnip{Arrhenius}{Life-acceleration model: temperature stress multiplies wear-out
  by $\exp[(E_a/k)(1/T_\mathrm{use}-1/T_\mathrm{stress})]$.}

\declaretermsnip{wavelengthLocker}{Etalon + dual-PD error signal that
  trims laser TEC or current onto a grid slot (common on discrete DFB/EML).}

\declaretermsnip{thermalCrosstalk}{Heater or self-heating on one ring
  shifts neighbors' resonances; worst case is full-bank traffic at max case $T$.}

\declaretermsnip{ACC}{Active copper cable: redriver (CTLE+VGA) inside cable; extends DAC reach.}

\declaretermsnip{AEC}{Active electrical cable: retimer (EQ+CDR) inside cable; longer twinax.}

\declaretermsnip{RLM}{Relative level mismatch: PAM4 level spacing uniformity. 1.0 is
  perfectly even; CEI often requires $\ge$0.95.}

\declaretermsnip{OMA}{Optical modulation amplitude: outer PAM4 level swing
  $P_1-P_0$.}

\declaretermsnip{COM}{Channel operating margin: statistical SNR of the fully equalized
  link. CEI go/no-go is typically COM $\ge$3~dB.}

\declaretermsnip{ISI}{Inter-symbol interference: energy from neighboring symbols
  overlapping the current unit interval.}

\declaretermsnip{BER}{Bit error ratio: errored bits divided by bits received.
  Pre-FEC BER is the validation metric; post-FEC is the service metric.}

\declaretermsnip{VNA}{Vector network analyzer: measures S-parameters versus
  frequency. Used for electrical and electro-optic bandwidth, return loss,
  and impedance match.}

\declaretermsnip{DFE}{Decision-feedback equalizer: cancels post-cursor ISI using past
  symbol decisions.}

\declaretermsnip{CTLE}{Continuous-time linear equalizer: analog high-frequency boost in
  SerDes or TIA.}

\declaretermsnip{FFE}{Feed-forward equalizer: FIR filter. TDECQ applies a bounded FFE as
  the reference equalizer.}

\declaretermsnip{SNDR}{Signal-to-noise-and-distortion ratio: transmitter quality ceiling
  including nonlinearity.}

\declaretermsnip{UI}{Unit interval: one symbol period ($=1/R_{\mathrm{sym}}$). Jitter
  specs are in UI.}

\declaretermsnip{PMD}{Physical medium dependent: the optical transceiver (Tx/Rx) in
  IEEE 802.3.}

\declaretermsnip{PCS}{Physical coding sublayer: 64B/66B line coding, 256B/257B
  transcoding, scrambling, and RS-FEC. Maps MAC data to the coded line stream.}

\declaretermsnip{PMA}{Physical medium attachment: serialization, lane multiplexing,
  and clock recovery. The SerDes lives here.}

\declaretermsnip{AUI}{Attachment unit interface: the electrical serial lanes between
  host and module (e.g.\ 400GAUI-4, at the faceplate cage).}

\declaretermsnip{xMII}{Media-independent interface: a wide parallel bus between MAC and
  PCS inside the chip (e.g.\ 400GMII).}

\declaretermsnip{LPO}{Linear pluggable optics: no retimer/DSP in the module; host SerDes
  drives the modulator directly.}

\declaretermsnip{LRO}{Linear retimed optics: retimer in the module but no heavy DSP;
  lighter than fully retimed pluggables.}

\declaretermsnip{XSR}{CEI extra-short reach: die-to-die or die-to-OE, mm-scale traces.}

\declaretermsnip{CPC}{Co-packaged copper: copper I/O at the package edge (OIF map);
  feeds faceplate cages or short copper without an on-package optical engine.}

\declaretermsnip{VSR}{CEI very short reach: host PCB plus one module connector
  ($\sim$200~mm).}

\declaretermsnip{MR}{CEI medium reach: longer host plus module channel than VSR.}

\declaretermsnip{LR}{CEI long reach: longest CEI copper class before optics take over.}

\declaretermsnip{FR8DR8}{802.3 800G single-mode reaches: FR8 = 2~km, DR8 = 500~m
  (eight lanes per direction).}

\declaretermsnip{SerDes}{Serializer/deserializer: high-speed I/O that maps parallel
  data to PAM4 on the wire.}

\declaretermsnip{dspserdes}{DSP-based SerDes: ADC on the Rx lane, digital FFE/DFE and
  timing recovery, DAC on Tx. Standard above $\sim$50~GBd; the ``DSP'' that cancels ISI
  at 224G lives inside the SerDes.}

\declaretermsnip{GBd}{Gigabaud: $10^9$ symbols per second. Line rate = baud
  $\times$ bits per symbol.}

\declaretermsnip{PHY}{Physical layer: PCS plus PMD; maps MAC frames to the line code.}

\declaretermsnip{DCA}{Digital communication analyzer: sampling oscilloscope used for
  eye diagrams and TDECQ.}

\declaretermsnip{CDR}{Clock-data recovery: extracts bit clock from the data stream;
  present in retimed modules, absent in LPO.}

\declaretermsnip{EML}{Electro-absorption modulated laser: DFB plus EAM on one InP chip;
  dominant 100--200G/lane pluggable Tx.}

\declaretermsnip{UEC}{Ultra Ethernet Consortium: open Ethernet stack for AI/HPC scale-out (UET transport, PHY extensions).}
\declaretermsnip{UALink}{Ultra Accelerator Link: open scale-up fabric for accelerator load/store (200G/lane class).}
\declaretermsnip{UCIe}{Universal Chiplet Interconnect Express: open die-to-die interconnect for chiplets on a package or interposer; parallel counterpart to CEI XSR.}

\declaretermsnip{OCS}{Optical circuit switch: a Layer-1 switch that steers light
  input-to-output with no O-E-O conversion, transparent to rate, format, and wavelength.
  Millisecond (MEMS) reconfiguration makes it a topology and failure-reroute switch, not
  a packet switch.}

\declaretermsnip{JESD47}{JEDEC's baseline IC stress-test qualification standard:
  temperature cycle, HTOL, HTSL, autoclave/HAST, mechanical shock and vibration.
  The silicon-side counterpart to Telcordia GR-468 for optoelectronics.}

\declaretermsnip{JESD78}{JEDEC IC latch-up test: overvoltage and $\pm100$~mA
  current-injection stress that classifies a CMOS/BiCMOS die's susceptibility to a
  parasitic-thyristor low-impedance latch.}

\declaretermsnip{AECQ100}{Automotive Electronics Council IC qualification standard:
  JEDEC stress methods plus tighter ESD targets and named temperature grades
  (Grade~3 to Grade~0). Not required for datacenter optics; useful as a borrowed
  signal when a driver/TIA die also ships an automotive part number.}

% --- Additional abbreviation margin snips ---

\declaretermsnip{APR}{Automatic power reduction: holds laser output at safe level on
  fiber break; probes for re-mate at Hazard Level 1M.}

\declaretermsnip{ALS}{Automatic laser shutdown: cuts laser power on fiber break.
  APR with restart preferred in modern systems (ITU-T G.664).}

\declaretermsnip{ASE}{Amplified spontaneous emission: broadband noise from optical
  amplifiers (SOA, EDFA). Adds to the receiver noise floor.}

\declaretermsnip{CW}{Continuous wave: unmodulated laser output. External modulators
  (MZM, ring, EAM) encode data onto a CW source.}

\declaretermsnip{CMOS}{Complementary metal-oxide-semiconductor: mainstream IC process;
  TIAs and SerDes at 16--3~nm nodes.}

\declaretermsnip{COD}{Catastrophic optical damage: sudden, irreversible facet failure
  in a laser under thermal or optical overstress.}

\declaretermsnip{COBO}{Consortium for On-Board Optics: standardized mid-board optical
  engines; largely leapfrogged by CPO in hyperscale.}

\declaretermsnip{COUPE}{Compact Universal Photonic Engine: TSMC's SoIC-X hybrid-bonded
  EIC-on-PIC packaging for CPO. Mass production 2026.}

\declaretermsnip{CXL}{Compute Express Link: coherent memory interconnect built on
  PCIe PHY (CXL 4.0 on PCIe 7.0). Potential optical reach.}

\declaretermsnip{DBR}{Distributed Bragg reflector: laser grating outside the gain
  region; enables tunable single-mode output.}

\declaretermsnip{DR}{Datacenter reach: IEEE 802.3 single-mode class, typically 500~m
  at 1310~nm. Default for AI scale-out optics.}

\declaretermsnip{DWDM}{Dense wavelength division multiplexing: channel spacing
  $\le$100~GHz; used in metro/long-haul and dense CPO WDM.}

\declaretermsnip{ELS}{External laser source: CW laser module feeding a co-packaged
  optical engine (superset of ELSFP/CW-WDM forms).}

\declaretermsnip{EO}{Electro-optic: conversion from voltage to optical phase or
  intensity. EO bandwidth sets modulator speed.}

\declaretermsnip{ER}{Extinction ratio: $P_1/P_0$ in dB. Higher ER widens the OMA for
  a given average power; trades against chirp and driver swing.}

\declaretermsnip{FAU}{Fiber array unit: a precision V-groove assembly that mates
  multiple fibers to a PIC edge or grating coupler array.}

\declaretermsnip{FIT}{Failures in time: failures per $10^9$ device-hours. Fleet FIT
  times link count sets expected failures per day.}

\declaretermsnip{FR}{Far reach: IEEE 802.3 single-mode class, 2~km. Tighter chirp and
  dispersion budget than DR.}

\declaretermsnip{FSR}{Free spectral range: wavelength span between adjacent resonances
  of a ring or etalon. Sets WDM channel packing density.}

\declaretermsnip{HBM}{High-bandwidth memory: stacked DRAM on interposer beside the
  accelerator die. Memory BW sets the decode roofline.}

\declaretermsnip{HCF}{Hollow-core fiber: guides light in air ($n\approx1$); 33\%
  lower latency than silica SMF. Loss record 0.091~dB/km (2025).}

\declaretermsnip{HTOL}{High-temperature operating life: accelerated stress test
  (GR-468). Arrhenius projects field wear-out from HTOL lots.}

\declaretermsnip{IBTA}{InfiniBand Trade Association: owns the InfiniBand Architecture
  spec (NDR 400G, XDR 800G/port); reuses QSFP/OSFP optics.}

\declaretermsnip{IEEE}{Institute of Electrical and Electronics Engineers: 802.3 owns
  Ethernet PHY/MAC, FEC (KP4), and optical PMD clauses.}

\declaretermsnip{MACsec}{IEEE 802.1AE link-layer encryption: line-rate confidentiality
  and integrity; transparent to PMD and LPO.}

\declaretermsnip{MEMS}{Micro-electro-mechanical systems: tilting mirror arrays in OCS;
  millisecond switching, $\sim$2~dB loss, 136$\times$136 radix.}

\declaretermsnip{MMF}{Multimode fiber: 850--940~nm VCSEL links (OM3/OM4/OM5); reach
  and modal BW limit to SR distances.}

\declaretermsnip{MoE}{Mixture of experts: transformer architecture that routes tokens
  to specialist sub-networks; all-to-all collective demand.}

\declaretermsnip{MPO}{Multi-fiber push-on: standard multi-fiber connector for parallel
  optics (8, 12, 16, 24, or 32 fibers per ferrule).}

\declaretermsnip{MRM}{Microring modulator: resonant Si modulator; compact, WDM-native,
  needs wavelength lock. CPO workhorse at 200G/ch.}

\declaretermsnip{MSA}{Multi-source agreement: industry spec for interoperable modules
  (e.g.\ LPO MSA, CW-WDM MSA, 100G Lambda MSA).}

\declaretermsnip{MZM}{Mach--Zehnder modulator: push-pull interferometer; broadband, low
  chirp. Si MZM, TFLN MZM, or hybrid platforms.}

\declaretermsnip{NFF}{No fault found: RMA unit passes all tests on return. High NFF
  rate points at triage or intermittent connector faults.}

\declaretermsnip{NIC}{Network interface card: host adapter connecting an accelerator or
  CPU to the scale-out fabric (Ethernet or IB).}

\declaretermsnip{OBO}{On-board optics: mid-board engine on host PCB (COBO standard);
  mostly bypassed for CPO in hyperscale deployments.}

\declaretermsnip{ORL}{Optical return loss: reflected power back toward the laser.
  Low ORL raises RIN and can seed burst errors.}

\declaretermsnip{OSA}{Optical spectrum analyzer: measures wavelength, SMSR, and
  side-mode structure of a laser source.}

\declaretermsnip{OSFP}{Octal Small Form-factor Pluggable: high-power faceplate cage for
  800G/1.6T/3.2T modules; larger thermal envelope than QSFP-DD.}

\declaretermsnip{PD}{Photodiode: converts photons to photocurrent. Ge-on-Si PIN is
  mainstream; APD adds gain; UTC for high saturation.}

\declaretermsnip{PIC}{Photonic integrated circuit: waveguides, modulators, detectors,
  and couplers on one chip (typically SOI at 220~nm Si).}

\declaretermsnip{PRBS}{Pseudo-random binary sequence: deterministic test pattern that
  exercises all bit transitions for BER and eye measurements.}

\declaretermsnip{PUE}{Power usage effectiveness: total facility power / IT load power.
  Typical hyperscale PUE 1.1--1.3; cooling overhead.}

\declaretermsnip{QSFPDD}{Quad Small Form-factor Pluggable Double Density: faceplate
  module cage for 400G/800G/1.6T with eight electrical lanes.}

\declaretermsnip{RDMA}{Remote direct memory access: zero-copy network transfer between
  accelerator memories; RoCE (Ethernet) or native IB.}

\declaretermsnip{SiGe}{Silicon-germanium BiCMOS: high-$f_T$ process for TIAs and
  modulator drivers at 100--200+~GHz bandwidth.}

\declaretermsnip{SiPh}{Silicon photonics: waveguides and modulators on SOI; CMOS
  fab-compatible. Mainstream for DR/FR and CPO.}

\declaretermsnip{SMF}{Single-mode fiber: 9~$\mu$m core; O-band (1310~nm) or C-band
  (1550~nm). G.652.D and G.657 for datacenter plant.}

\declaretermsnip{SOA}{Semiconductor optical amplifier: inline optical gain (III-V);
  adds ASE noise. Used as preamplifier for sensitivity.}

\declaretermsnip{SOI}{Silicon on insulator: substrate for silicon photonics; 220~nm Si
  layer on buried oxide confines the optical mode.}

\declaretermsnip{SPC}{Statistical process control: lot-by-lot control charts
  ($I_\mathrm{th}$, SMSR, power) to catch drift between qual stress tests.}

\declaretermsnip{SR}{Short reach: multimode fiber class (typically 100~m over OM4);
  VCSEL at 850~nm. Legacy rack optics.}

\declaretermsnip{TCO}{Total cost of ownership: acquisition (BOM, yield) plus lifetime
  energy plus service (FIT $\times$ replacement cost).}

\declaretermsnip{TEC}{Thermoelectric cooler: Peltier device that holds laser junction
  or ring at a controlled temperature.}

\declaretermsnip{TFLN}{Thin-film lithium niobate: sub-$\mu$m LN on Si handle.
  $>$110~GHz EO BW; leading path to native 224~GBd PAM4.}

\declaretermsnip{TP}{Test point: reference plane where a measurement is taken (TP0
  through TP5, plus TP1a and TP4a for host-referenced).}

\declaretermsnip{TRO}{Transmit retimed optic: module with DSP only on Tx path;
  Rx is linear. Symmetric opposite of LRO.}

\declaretermsnip{TWE}{Traveling-wave electrode: RF transmission line on a modulator
  that velocity-matches the optical mode for high BW.}

\declaretermsnip{TWI}{Two-wire interface: I2C-like serial bus for CMIS module
  management (SCL + SDA).}

\declaretermsnip{UTC}{Uni-traveling-carrier photodiode: electron-only transport;
  $>$200~GHz BW and high saturation current. Linear/LPO niche.}

\declaretermsnip{VGA}{Variable gain amplifier: adjustable-gain stage in a redriver or
  TIA AGC path.}

\declaretermsnip{VOA}{Variable optical attenuator: calibrated loss element for
  sensitivity sweeps and stressed-receiver testing.}

\declaretermsnip{WDM}{Wavelength division multiplexing: multiple $\lambda$ on one
  fiber. CWDM (20~nm spacing) or DWDM ($\le$100~GHz).}

\declaretermsnip{XPO}{eXtra-dense Pluggable Optics: liquid-cooled 12.8~Tb/s pluggable
  (Arista MSA, OFC 2026); front-panel serviceable at CPO density.}

\declaretermsnip{DDM}{Digital diagnostic monitoring: per-lane telemetry in CMIS
  (Tx/Rx power, bias, temperature, LOS/LOL flags).}

\declaretermsnip{MAC}{Media access control: Ethernet data-link sublayer; builds frames,
  addresses, CRC. MAC rate is payload throughput.}

\declaretermsnip{BOM}{Bill of materials: component and assembly cost of a module or
  system. DSP presence is a large BOM driver.}

\declaretermsnip{BERT}{Bit-error-ratio tester: instrument or ASIC function that
  generates PRBS and counts errors for pre-/post-FEC BER.}

\declaretermsnip{BiCMOS}{Bipolar CMOS: process combining bipolar (high $f_T$) and
  CMOS on one die; used for high-speed TIAs and drivers.}

\declaretermsnip{CWWDM}{CW-WDM MSA: multi-source agreement for multi-wavelength CW
  laser sources feeding WDM photonic engines (CPO).}

\declaretermsnip{InP}{Indium phosphide: III-V semiconductor for lasers, EAMs, SOAs,
  and high-speed photodiodes in the 1310/1550~nm bands.}

\declaretermsnip{OCS}{Optical circuit switch: Layer-1 switch steering light with no
  O-E-O conversion; transparent to rate/format/wavelength. MEMS OCS: ms switching,
  $\sim$2~dB loss, topology reconfiguration.}

\declaretermsnip{PIN}{PIN photodiode: p-intrinsic-n structure; no internal gain.
  Ge-on-Si waveguide PIN is the mainstream short-reach detector.}

\declaretermsnip{PSRR}{Power-supply rejection ratio: how well a bias board or driver
  rejects rail noise. Weak PSRR converts electrical noise into optical intensity
  noise through the laser.}

\declaretermsnip{MPI}{Multipath interference: coherent beating from two or more
  reflective paths. Produces a BER floor and clustered FEC errors that more
  launch power does not fix.}

\declaretermsnip{PAM4}{Four-level pulse-amplitude modulation: two bits per symbol.
  Short-reach 100--200G/lane links use PAM4 with KP4 FEC.}
